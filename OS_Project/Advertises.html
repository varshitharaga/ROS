import subprocess
import yaml

def advertise_topic(topic_name, message_type, message):
    try:
        # Convert message to YAML format
        message_yaml = yaml.dump({'data': message})
        
        # Run the 'ros2 topic pub' command to advertise the topic
        subprocess.run(['ros2', 'topic', 'pub', topic_name, message_type, message_yaml], check=True)
        print(f"Topic '{topic_name}' advertised successfully.")
    except subprocess.CalledProcessError as e:
        print("Error:", e)
