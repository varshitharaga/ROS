                import time
                
                class Task:
                    def _init_(self, name, deadline):
                        self.name = name
                        self.deadline = deadline
                        self.start_time = None
                        self.end_time = None
                
                class EDFScheduler:
                    def _init_(self):
                        self.tasks = [
                            Task('Task1', time.time() + 5.0),
                            Task('Task2', time.time() + 10.0),
                            Task('Task3', time.time() + 8.0)
                        ]
                        self.gantt_chart = []
                
                    def execute_tasks(self):
                        current_time = time.time()
                        # Sort tasks based on deadlines
                        self.tasks.sort(key=lambda x: x.deadline)
                        for task in self.tasks:
                            start_time = max(current_time, task.deadline - 5.0)  # Ensuring tasks start at their deadline
                            end_time = start_time + 1.0  # Assuming each task takes 1 unit of time
                            task.start_time = start_time
                            task.end_time = end_time
                            current_time = end_time
                
                    def generate_gantt_chart(self):
                        for task in self.tasks:
                            self.gantt_chart.append((task.name, task.start_time, task.end_time))
                
                    def print_gantt_chart(self):
                        print("Gantt Chart:")
                        print("------------")
                        for entry in self.gantt_chart:
                            print(f"{entry[0]}: {entry[1]:.2f} - {entry[2]:.2f}")
                
                def main():
                    scheduler = EDFScheduler()
                    scheduler.execute_tasks()
                    scheduler.generate_gantt_chart()
                    scheduler.print_gantt_chart()
                
                if _name_ == '_main_':
                    main()
