            #!/usr/bin/env python3

import subprocess

def get_node_names():
    try:
        # Run the 'ros2 node list' command and capture its output
        result = subprocess.run(['ros2', 'node', 'list'], capture_output=True, text=True, check=True)
        # Split the output by newline characters to get individual node names
        node_names = result.stdout.strip().split('\n')
        return node_names
    except subprocess.CalledProcessError as e:
        print("Error:", e)
        return []

if _name_ == "_main_":
    node_names = get_node_names()
    print("Active nodes in the ROS 2 system:")
    for node_name in node_names:
        print("- Node:", node_name)
