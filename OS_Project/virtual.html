            import random

            class Process:
                def _init_(self, name, memory_required):
                    self.name = name
                    self.memory_required = memory_required
            
            class VirtualMemoryManager:
                def _init_(self, total_memory):
                    self.total_memory = total_memory
                    self.processes = []
                    self.free_memory = total_memory
            
                def allocate_memory(self, process):
                    if process.memory_required <= self.free_memory:
                        self.processes.append(process)
                        self.free_memory -= process.memory_required
                        print(f'Allocated {process.memory_required} memory to {process.name}')
                    else:
                        print(f'Not enough memory available to allocate {process.memory_required} to {process.name}')
            
                def deallocate_memory(self, process_name):
                    for proc in self.processes:
                        if proc.name == process_name:
                            self.free_memory += proc.memory_required
                            self.processes.remove(proc)
                            print(f'Deallocated memory from {proc.name}')
                            return
                    print(f'Process {process_name} not found in the allocated processes')
            
            def main():
                vm_manager = VirtualMemoryManager(total_memory=1000)
            
                # Example usage
                process1 = Process('Process1', random.randint(50, 200))
                process2 = Process('Process2', random.randint(50, 200))
                process3 = Process('Process3', random.randint(50, 200))
            
                vm_manager.allocate_memory(process1)
                vm_manager.allocate_memory(process2)
                vm_manager.deallocate_memory('Process1')
                vm_manager.allocate_memory(process3)
            
            if _name_ == '_main_':
                main()
